450 static inline int cose_sign_set_ct(cose_sign_t *sign, int32_t value)
451 {
452  int res = COSE_OK;
453  cose_hdr_t *hdr = cose_hdr_get(sign->hdrs, COSE_SIGN_HDR_MAX, COSE_HDR_CONTENT_TYPE);
454 
455  if (!hdr) {
456  res = cose_sign_add_hdr_value(sign, COSE_HDR_CONTENT_TYPE, COSE_HDR_FLAGS_PROTECTED, value);
457  }
458  else {
459  hdr->v.value = value;
460  hdr->flags |= COSE_HDR_FLAGS_PROTECTED;
461  hdr->type = COSE_HDR_TYPE_INT;
462  }
463  return res;
464 }
